# ipcalc

a simple IPv4 calculator

#################################################
#                                               #
# syntax:                                       #
#                                               #
# ./ipcalc.sh [ip-address]/[cidr]               #
# ./ipcalc.sh 192.168.0.1/24                    #
#                                               #
# or                                            #
#                                               #
# ./ipcalc.sh [ip-address] [netmask]            #
# ./ipcalc.sh 192.168.0.1 255.255.255.0         #
#                                               #
#################################################


example

/ipcalc.sh 192.168.0.123/24
